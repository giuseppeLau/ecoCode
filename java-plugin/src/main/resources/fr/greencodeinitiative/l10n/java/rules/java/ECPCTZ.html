In loops, prefer comparison to zero conditions.

Noncompliant For Loop Solution

    public class nonCompliantForSolution {
        public static void main(String[] args) {
        
			for (int i = 0; i < 5; i++) { // Noncompliant
				// Do Something;
			}
		}    
    }

Compliant For Loop Solution

    public class compliantForSolution {
        public static void main(String[] args) {
        
			for (int i = 5; i > 0; i--) { // Compliant
				// Do Something;
			}
		}    
    }

Noncompliant While Loop Solution

    public class nonCompliantForSolution {
        public static void main(String[] args) {
        
			int i = 0;
			while(i < 5) { // Noncompliant
				// Do Something
				i++;
			}
		}    
    }

Compliant While Loop Solution

    public class compliantForSolution {
        public static void main(String[] args) {
        
			int i = 5;
			while(i > 0) { // Noncompliant
				// Do Something
				i--;
			}
		}    
    }
